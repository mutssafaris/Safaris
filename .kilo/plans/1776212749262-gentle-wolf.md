```javascript
var hotelImages = { main: '', gallery: [] };

// Initialize image upload
ImageUploadComponent.create(document.getElementById('hotel-image-upload'), {
    fieldId: 'hotel-main-image',
    multiple: false,
    value: [],
    onImageAdd: function(src, data) {
        hotelImages.main = data.url || data.data || src;
        updatePreview('image-preview', hotelImages.main);
    },
    onImageRemove: function(data) {
        hotelImages.main = '';
    }
});

// Update populateForm to show existing image
function populateForm(hotel) {
    // ... existing fields ...
    if (hotel.image) {
        hotelImages.main = hotel.image;
        // If it's a URL, show in preview
        if (hotel.image.startsWith('http')) {
            updatePreview('image-preview', hotel.image);
        }
    }
}
```

### Step 2: Add Image Upload to Save Functions
Update saveChanges() and publishChanges() to handle image uploads:

```javascript
function saveChanges() {
    var formData = getFormData();
    var hotelId = document.getElementById('hotel-id').value;
    showToast('Saving changes...');
    
    // Upload image if changed
    var uploadPromises = [];
    if (hotelImages.main && !hotelImages.main.startsWith('http')) {
        uploadPromises.push(
            ImageService.uploadImage(hotelImages.main, { category: 'hotels' })
                .then(function(result) { formData.image = result.url; })
        );
    } else {
        formData.image = hotelImages.main;
    }
    
    Promise.all(uploadPromises).then(function() {
        return ManagerService.updateHotel(hotelId, formData);
    }).then(function() {
        showToast('Changes saved successfully');
        setTimeout(function() { window.location.href = 'list.html'; }, 1000);
    }).catch(function(err) {
        showToast('Failed to save changes', 'error');
    });
}
```
